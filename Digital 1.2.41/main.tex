%% -------------------------------------------------------
\begin{frame}{Question}
The solution of
\begin{align}
    ye^x\,dx + \brak{4y + e^x}\,dy = 0, \quad y(0) = -1
\end{align}
is
\vspace{0.3cm}
\begin{columns}
\column{0.5\textwidth}
    \textbf{a)} $ye^x + 2y^2 - 1 = 0$\\[0.4cm]
    \textbf{b)} $e^x + y^2 - 2 = 0$
\column{0.5\textwidth}
    \textbf{c)} $ye^x - y^2 = 0$\\[0.4cm]
    \textbf{d)} $ye^x + y^2 - 1 = 0$
\end{columns}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Rearrange by dividing throughout by $dy$:
\begin{align}
    ye^x\,\frac{dx}{dy} + 4y + e^x &= 0 \\
    ye^x\,\frac{dx}{dy} + e^x &= -4y
\end{align}

Observe the product rule for differentiation with respect to $y$:
\begin{align}
    \frac{d}{dy}\brak{ye^x} = e^x + ye^x\,\frac{dx}{dy} \label{eq:prod}
\end{align}

The left-hand side is exactly $\dfrac{d}{dy}\brak{ye^x}$, so:
\begin{align}
    \frac{d}{dy}\brak{ye^x} = -4y
\end{align}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Integrating both sides with respect to $y$:
\begin{align}
    \int \frac{d}{dy}\brak{ye^x}\,dy &= \int -4y\,dy \\
    ye^x &= -2y^2 + C \label{eq:gen}
\end{align}

Applying the initial condition $y(0) = -1$, i.e.\ $x=0$ when $y=-1$:
\begin{align}
    \brak{-1}e^{0} &= -2\brak{-1}^2 + C \\
    -1 &= -2 + C \\
    C &= 1
\end{align}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Substituting $C = 1$ into \eqref{eq:gen}:
\begin{align}
    ye^x &= -2y^2 + 1 \\
    \boxed{ye^x + 2y^2 - 1 = 0}
\end{align}

\textbf{Verification:} At $y(0)=-1$,
\begin{align}
    \brak{-1}e^{0} + 2\brak{-1}^2 - 1 = -1 + 2 - 1 = 0 \quad \checkmark
\end{align}

\begin{block}{Answer}
\centering \textbf{Option (a):} $ye^x + 2y^2 - 1 = 0$
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{How the Numerical (odeint) Plot is Obtained}
The implicit ODE $ye^x\,dx + (4y+e^x)\,dy = 0$ is rewritten as an explicit $\dfrac{dy}{dx}$:
\begin{align}
    \frac{dy}{dx} = \frac{-ye^x}{4y + e^x} \label{eq:odeint}
\end{align}

\texttt{scipy.integrate.odeint} solves this using the \textbf{LSODA} algorithm:
\begin{itemize}
    \item Starts at the initial condition $y(0) = -1$
    \item At each step $x_k$, evaluates $\frac{dy}{dx}\big|_{x_k,\,y_k}$ from \eqref{eq:odeint}
    \item Advances: $y_{k+1} \approx y_k + h\cdot\frac{dy}{dx}\big|_{x_k,\,y_k}$ using adaptive Runge-Kutta / Adams methods
    \item Automatically shrinks step size $h$ where curve bends sharply
\end{itemize}

\begin{block}{Note}
    Accuracy degrades when $4y + e^x \to 0$ (denominator $\to 0$, slope $\to \infty$).
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{How the Analytical Plot is Obtained}
The implicit solution $ye^x + 2y^2 - 1 = 0$ is a quadratic in $y$:
\begin{align}
    2y^2 + e^x\cdot y - 1 = 0
\end{align}

Applying the quadratic formula with $A=2,\ B=e^x,\ C=-1$:
\begin{align}
    y = \frac{-e^x \pm \sqrt{e^{2x} + 8}}{4}
\end{align}

Select the $\boldsymbol{-}$ branch to match $y(0)=-1$:
\begin{align}
    y\big|_{x=0} = \frac{-1 - \sqrt{1+8}}{4} = \frac{-1-3}{4} = -1 \quad \checkmark
\end{align}

\begin{block}{Closed-form used for plotting}
\[
    y(x) = \frac{-e^x - \sqrt{e^{2x}+8}}{4}
\]
\end{block}
This is exact for all $x$ — no accumulation of numerical error.
\end{frame}

\frametitle{Python Code: Setup and Numerical Solution}
\begin{lstlisting}
import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint


def ode(y, x):
    ex = np.exp(x)
    denom = 4*y + ex
    return -y * ex / denom

x = np.linspace(0, 2, 500)
sol = odeint(ode, [-1.0], x)
\end{lstlisting}
\end{frame}

%% -------------------------------------------------------
\begin{frame}[fragile]

def y_analytical(x_vals):
    ex = np.exp(x_vals)
    return (-ex - np.sqrt(ex**2 + 8)) / 4

y_anal = y_analytical(x)

print(f"Analytical y(0) = {y_analytical(np.array([0.0]))[0]:.4f}")

\end{lstlisting}
\end{frame}
