\begin{frame}{Question}
If $u(t)$, $r(t)$ denote the unit step and unit ramp functions respectively and $u(t)*r(t)$ their convolution, then the function $u(t+1)*r(t-2)$ is given by
\vspace{0.5cm}
\begin{columns}
\column{0.5\textwidth}
    \textbf{a)} $\dfrac{1}{2}(t-1)(t-2)$\\[0.5cm]
    \textbf{b)} $\dfrac{1}{2}(t-1)(t-2)$
\column{0.5\textwidth}
    \textbf{c)} $\dfrac{1}{2}(t-1)^2\,u(t-1)$\\[0.5cm]
    \textbf{d)} None of the above
\end{columns}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Recall the definitions of the signals:
\begin{align}
    u(t) &= \begin{cases} 1, & t \geq 0 \\ 0, & t < 0 \end{cases}
    \implies
    u(t+1) = \begin{cases} 1, & t \geq -1 \\ 0, & t < -1 \end{cases} \\[0.4cm]
    r(t) &= t\,u(t) = \begin{cases} t, & t \geq 0 \\ 0, & t < 0 \end{cases}
    \implies
    r(t-2) = \begin{cases} t-2, & t \geq 2 \\ 0, & t < 2 \end{cases}
\end{align}

The convolution integral is:
\begin{align}
    y(t) = u(t+1)*r(t-2) = \int_{-\infty}^{\infty} u(\tau+1)\,r(t-\tau-2)\;d\tau
\end{align}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Both factors must be nonzero simultaneously:
\begin{align}
    u(\tau+1) \neq 0 \quad &\Rightarrow \quad \tau \geq -1 \label{eq:c1}\\
    r(t-\tau-2) \neq 0 \quad &\Rightarrow \quad t-\tau-2 \geq 0 \quad \Rightarrow \quad \tau \leq t-2 \label{eq:c2}
\end{align}

For a non-trivial integral, we need $-1 \leq \tau \leq t-2$, which requires:
\begin{align}
    -1 \leq t-2 \quad \Rightarrow \quad t \geq 1
\end{align}

\begin{block}{}
$y(t) = 0$ for $t < 1$. \quad The convolution output starts at $t=1$.
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
For $t \geq 1$, substituting $u(\tau+1)=1$ over $\sbrak{-1,\,t-2}$:
\begin{align}
    y(t) &= \int_{-1}^{t-2} \brak{t - \tau - 2}\;d\tau
\end{align}

Let $s = t-\tau-2$, so $ds = -d\tau$.\\
When $\tau=-1$: $s = t-1$. \quad When $\tau=t-2$: $s = 0$.
\begin{align}
    y(t) &= \int_{t-1}^{0} s\cdot(-ds) = \int_{0}^{t-1} s\;ds \\
    y(t) &= \left[\frac{s^2}{2}\right]_0^{t-1} = \frac{\brak{t-1}^2}{2}
\end{align}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Solution}
Combining both cases:
\begin{align}
    y(t) = u(t+1)*r(t-2) = \begin{cases} \dfrac{(t-1)^2}{2}, & t \geq 1 \\[6pt] 0, & t < 1 \end{cases}
\end{align}

This is written compactly as:
\begin{align}
    \boxed{u(t+1)*r(t-2) = \frac{1}{2}\brak{t-1}^2\,u(t-1)}
\end{align}

\begin{block}{Answer}
\centering \textbf{Option (c):} $\dfrac{1}{2}(t-1)^2\,u(t-1)$
\end{block}
\end{frame}

\begin{frame}[fragile]
\frametitle{Python Code}
\begin{lstlisting}
import numpy as np
import matplotlib.pyplot as plt

# Define unit step
def u(t):
    return np.where(t >= 0, 1.0, 0.0)

# Define unit ramp
def r(t):
    return t * u(t)

# Time axis
t = np.linspace(-1, 6, 10000)
dt = t[1] - t[0]
\end{lstlisting}
\end{frame}

%% -------------------------------------------------------
\begin{frame}[fragile]
\frametitle{Python Code}
\begin{lstlisting}
# Signals to convolve
f1 = u(t + 1)       # u(t+1)
f2 = r(t - 2)       # r(t-2)

# Convolution (numerical)
y_num = np.convolve(f1, f2, mode='full') * dt
t_conv = np.linspace(2*t[0], 2*t[-1], len(y_num))

# Analytical result
y_anal = 0.5 * (t - 1)**2 * u(t - 1)
\end{lstlisting}
\end{frame}

%% -------------------------------------------------------
\begin{frame}[fragile]
\frametitle{Python Code}
\begin{lstlisting}
# Plot
plt.figure(figsize=(8, 4))
plt.plot(t_conv, y_num, 'b-', label='Numerical convolution')
plt.plot(t, y_anal, 'r--', label=r'Analytical: $\frac{1}{2}(t-1)^2 u(t-1)$')
plt.xlim(-1, 6)
plt.ylim(-0.5, 8)
plt.xlabel('t')
plt.ylabel('y(t)')
plt.title(r'$u(t+1) * r(t-2)$')
plt.legend()

plt.savefig("convolution.png", dpi=300)
plt.show()
\end{lstlisting}
