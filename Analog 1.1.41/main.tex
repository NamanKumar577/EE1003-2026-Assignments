%% -------------------------------------------------------
\begin{frame}{Question}
If $u(t)$, $r(t)$ denote the unit step and unit ramp functions respectively and $u(t)*r(t)$ their convolution, then the function $u(t+1)*r(t-2)$ is given by
\vspace{0.5cm}
\begin{columns}
\column{0.5\textwidth}
    \textbf{a)} $\dfrac{1}{2}(t-1)(t-2)$\\[0.5cm]
    \textbf{b)} $\dfrac{1}{2}(t-1)(t-2)$
\column{0.5\textwidth}
    \textbf{c)} $\dfrac{1}{2}(t-1)^2\,u(t-1)$\\[0.5cm]
    \textbf{d)} None of the above
\end{columns}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{General Setup: Shifted Signals}
Define general time-shifted unit step and unit ramp:
\begin{align}
    u(t-a) &= \begin{cases} 1, & t \geq a \\ 0, & t < a \end{cases} \\[0.3cm]
    r(t-b) &= \brak{t-b}\,u(t-b) = \begin{cases} t-b, & t \geq b \\ 0, & t < b \end{cases}
\end{align}

We compute the general convolution:
\begin{align}
    y(t) = u(t-a) * r(t-b) = \int_{-\infty}^{\infty} u(\tau - a)\,r(t-\tau-b)\;d\tau
\end{align}

\begin{block}{Goal}
    Derive a closed-form for any $a,b \in \mathbb{R}$, then substitute $a=-1,\ b=2$.
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{General Solution: Limits of Integration}
Both factors must be nonzero simultaneously:
\begin{align}
    u(\tau - a) \neq 0 \quad &\Rightarrow \quad \tau \geq a \label{eq:g1}\\
    r(t-\tau-b) \neq 0 \quad &\Rightarrow \quad t-\tau-b \geq 0 \quad \Rightarrow \quad \tau \leq t-b \label{eq:g2}
\end{align}

Active integration range: $a \leq \tau \leq t-b$, which requires:
\begin{align}
    a \leq t - b \quad \Rightarrow \quad t \geq a + b
\end{align}

\begin{block}{}
    $y(t) = 0$ for $t < a+b$. \quad Output starts at $t = a+b$.
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{General Solution: Evaluating the Integral}
For $t \geq a+b$, with $u(\tau-a)=1$ over $\sbrak{a,\,t-b}$:
\begin{align}
    y(t) &= \int_{a}^{t-b} \brak{t - \tau - b}\;d\tau
\end{align}

Let $s = t - \tau - b$, so $ds = -d\tau$:
\begin{itemize}
    \item When $\tau = a$: \quad $s = t - a - b$
    \item When $\tau = t-b$: \; $s = 0$
\end{itemize}
\begin{align}
    y(t) &= \int_{t-a-b}^{0} s\cdot(-ds) = \int_{0}^{t-a-b} s\;ds \\
    y(t) &= \left[\frac{s^2}{2}\right]_0^{t-a-b} = \frac{\brak{t-a-b}^2}{2}
\end{align}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{General Result}
Combining both cases, the \textbf{general convolution formula} is:
\begin{align}
    \boxed{u(t-a) * r(t-b) = \frac{1}{2}\brak{t-a-b}^2\,u(t-a-b)}
\end{align}

\vspace{0.2cm}
\textbf{Key observations:}
\begin{itemize}
    \item The output is always a \textbf{quadratic ramp} — same shape as $u(t)*r(t)$
    \item It is simply the base case $\frac{1}{2}t^2 u(t)$ \textbf{shifted right by} $a+b$
    \item $a < 0$ (step advanced) \textbf{reduces} the onset time
    \item $b > 0$ (ramp delayed) \textbf{increases} the onset time
    \item Net shift $= a + b$
\end{itemize}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Applying to the Question}
Substitute $a = -1$ (since $u(t+1) = u(t-(-1))$) and $b = 2$:
\begin{align}
    u(t+1) * r(t-2) &= \frac{1}{2}\brak{t-(-1)-2}^2\,u\brak{t-(-1)-2} \\
                    &= \frac{1}{2}\brak{t-1}^2\,u(t-1)
\end{align}

Net shift $= a + b = -1 + 2 = +1$ \quad (output starts at $t=1$)

\begin{block}{Answer}
\centering \textbf{Option (c):} $\dfrac{1}{2}(t-1)^2\,u(t-1)$
\end{block}
\end{frame}

%% -------------------------------------------------------
\begin{frame}{Relation Between the Three Cases}
\begin{center}
\renewcommand{\arraystretch}{1.6}
\begin{tabular}{|c|c|c|c|}
\hline
\textbf{Case} & $a$ & $b$ & \textbf{Result} \\
\hline
Base: $u(t)*r(t)$ & $0$ & $0$ & $\frac{1}{2}t^2\,u(t)$, onset $t=0$ \\
\hline
General: $u(t-a)*r(t-b)$ & $a$ & $b$ & $\frac{1}{2}(t{-}a{-}b)^2\,u(t{-}a{-}b)$ \\
\hline
Question: $u(t+1)*r(t-2)$ & $-1$ & $2$ & $\frac{1}{2}(t-1)^2\,u(t-1)$, onset $t=1$ \\
\hline
\end{tabular}
\end{center}

\vspace{0.2cm}
All three are \textbf{identical parabolic shapes} -- only the starting point on the $t$-axis differs by $a+b$.
The question's curve is the base case $u(t)*r(t)$ shifted \textbf{right by 1 unit}.
\end{frame}

import numpy as np
import matplotlib.pyplot as plt

def u(t):
    return np.where(t >= 0, 1.0, 0.0)

t = np.linspace(-2, 8, 10000)

y_base = 0.5 * t**2 * u(t)

a, b = -1, 2
y_general = 0.5 * (t - a - b)**2 * u(t - a - b)

y_question = 0.5 * (t - 1)**2 * u(t - 1)

fig, axes = plt.subplots(2, 1, figsize=(9, 9), sharex=True)
\end{lstlisting}
